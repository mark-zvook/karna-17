# KARNA-17 — RTC overlay fragment for /boot/firmware/config.txt

# Chip: RV-3028-C7 (I2C addr 0x52, no collision with MPU-6500 @ 0x68 or INA219 @ 0x40)
# Kernel driver: rtc-rv3028 (mainline since 5.7)
#
# Append these lines to /boot/firmware/config.txt (or include via setup_rtc.sh).
# Do NOT add dtoverlay=i2c-rtc,ds3231 — that chip is 0x68 and collides with IMU.

# Enable I2C bus 1 (already enabled for IMU/INA219 — keep if present)
dtparam=i2c_arm=on
dtparam=i2c_arm_baudrate=100000

# Load RTC driver for RV-3028 on I2C-1
dtoverlay=i2c-rtc,rv3028
